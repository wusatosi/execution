56 template <typename Error>

58 this->state->error = ::beman::execution::detail::as_except_ptr(::std::forward<Error>(error));
59 this->state->loop.finish();

77struct sync_wait_t {
78 template <typename Sender>
79 auto apply_sender(Sender&& sender) const {
80 ::beman::execution::detail::sync_wait_state<Sender> state;
81 auto op{::beman::execution::connect(::std::forward<Sender>(sender),
82 ::beman::execution::detail::sync_wait_receiver<Sender>{&state})};
83 ::beman::execution::start(op);
84
85 state.loop.run();
86 if (state.error) {
87 ::std::rethrow_exception(state.error);
88 }
89 return ::std::move(state.result);
90 }
