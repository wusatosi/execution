27template <class Sndr, class Promise>
28class sender_awaitable {
29 struct unit {};
30 using value_type =
31 ::beman::execution::detail::single_sender_value_type<Sndr, ::beman::execution::env_of_t<Promise>>;
32 using result_type = ::std::conditional_t<::std::is_void_v<value_type>, unit, value_type>;
33 using variant_type = ::std::variant<::std::monostate, result_type, ::std::exception_ptr>;
34 struct awaitable_receiver {
35 using receiver_concept = ::beman::execution::receiver_t;
36
37 template <class... Args>
38 requires ::std::constructible_from<result_type, Args...>
39 void set_value(Args&&... args) && noexcept {
40 try {
41 result_ptr_->template emplace<1>(::std::forward<Args>(args)...);
42 } catch (...) {
43 result_ptr_->template emplace<2>(::std::current_exception());
44 }
45 continuation_.resume();
46 }
47
48 template <class Error>
49 void set_error(Error&& error) && noexcept {
50 result_ptr_->template emplace<2>(::beman::execution::detail::as_except_ptr(::std::forward<Error>(error)));
51 continuation_.resume();
52 }
53
54 void set_stopped() && noexcept {
55 static_cast<::std::coroutine_handle<>>(continuation_.promise().unhandled_stopped()).resume();
56 }
57
58 auto get_env() const noexcept {
59 return ::beman::execution::detail::fwd_env{::beman::execution::get_env(continuation_.promise())};
60 }
61
62 variant_type* result_ptr_;
63 ::std::coroutine_handle<Promise> continuation_;
64 };
65 using op_state_type = ::beman::execution::connect_result_t<Sndr, awaitable_receiver>;
66
67 variant_type result{};
68 op_state_type state;
69
70 public:
71 sender_awaitable(Sndr&& sndr, Promise& p)
72 : state{::beman::execution::connect(
73 ::std::forward<Sndr>(sndr),
74 awaitable_receiver{::std::addressof(result), ::std::coroutine_handle<Promise>::from_promise(p)})} {}
75
76 static constexpr bool await_ready() noexcept { return false; }
77 void await_suspend(::std::coroutine_handle<Promise>) noexcept { ::beman::execution::start(state); }
78 value_type await_resume() {
79 if (::std::holds_alternative<::std::exception_ptr>(result)) {
80 ::std::rethrow_exception(::std::get<::std::exception_ptr>(result));
81 }
82 if constexpr (::std::is_void_v<value_type>) {
83 return;
84 } else {
85 return ::std::get<value_type>(result);
86 }
87 }
88};
