49template <typename Completion>
50struct let_t {
51 template <::beman::execution::detail::movable_value Fun>
52 auto operator()(Fun&& fun) const {
53 return ::beman::execution::detail::sender_adaptor{*this, ::std::forward<Fun>(fun)};
54 }
55 template <::beman::execution::sender Sender, ::beman::execution::detail::movable_value Fun>
56 auto operator()(Sender&& sender, Fun&& fun) const {
57 auto domain(::beman::execution::detail::get_domain_early(sender));
58 return ::beman::execution::detail::transform_sender(
59 domain,
60 ::beman::execution::detail::make_sender(*this, ::std::forward<Fun>(fun), std::forward<Sender>(sender)));
61 }
62
63 template <typename Sender>
64 static auto env(Sender&& sender) {
65 if constexpr (requires {
66 ::beman::execution::detail::sched_env(
67 ::beman::execution::get_completion_scheduler<Completion>(
68 ::beman::execution::get_env(sender)));
69 })
70 return ::beman::execution::detail::sched_env(
71 ::beman::execution::get_completion_scheduler<Completion>(::beman::execution::get_env(sender)));
72 else if constexpr (requires {
73 ::beman::execution::detail::make_env(
74 ::beman::execution::get_domain,
75 ::beman::execution::get_domain(::beman::execution::get_env(sender)));
76 })
77 return ::beman::execution::detail::make_env(
78 ::beman::execution::get_domain, ::beman::execution::get_domain(::beman::execution::get_env(sender)));
79 else
80 return ::beman::execution::empty_env{};
81 }
82 template <typename Sender, typename Env>
83 static auto join_env(Sender&& sender, Env&& e) -> decltype(auto) {
84 return ::beman::execution::detail::join_env(env(sender), ::beman::execution::detail::fwd_env(e));
85 }

133 static constexpr auto get_state{[]<typename Sender, typename Receiver>(Sender&& sender, Receiver&& receiver) {
134 auto& fun{sender.template get<1>()};
135 auto& child{sender.template get<2>()};

137 using fun_t = ::std::remove_cvref_t<decltype(fun)>;
138 using child_t = ::std::remove_cvref_t<decltype(child)>;
139 using env_t = decltype(::beman::execution::detail::let_t<Completion>::env(child));
140 using sigs_t = ::beman::execution::completion_signatures_of_t<child_t, ::beman::execution::env_of_t<Receiver>>;
141 using comp_sigs_t = ::beman::execution::detail::meta::filter<filter_pred, sigs_t>;
142 using type_list_t = ::beman::execution::detail::meta::to<::std::variant, comp_sigs_t>;
143 using tuples_t = ::beman::execution::detail::meta::transform<to_tuple_t, type_list_t>;
144 using unique_t = ::beman::execution::detail::meta::unique<tuples_t>;
145 using args_t = ::beman::execution::detail::meta::prepend<std::monostate, unique_t>;
146 using ops_t = ::beman::execution::detail::meta::prepend<
147 ::std::monostate,
148 ::beman::execution::detail::meta::unique<::beman::execution::detail::meta::transform<
149 to_state<fun_t, ::std::remove_cvref_t<Receiver>, env_t>::template trans,

164 template <typename Receiver, typename... Args>
165 static auto let_bind(auto& state, Receiver& receiver, Args&&... args) {
166 using args_t = ::beman::execution::detail::decayed_tuple<Args...>;
167 auto mkop{[&] {
168 return ::beman::execution::connect(
169 ::std::apply(::std::move(state.fun),
170 state.args.template emplace<args_t>(::std::forward<Args>(args)...)),
171 let_receiver<Receiver, decltype(state.env)>{receiver, state.env});
172 }};
173 ::beman::execution::start(
174 state.ops2.template emplace<decltype(mkop())>(beman::execution::detail::emplace_from{mkop}));
175 }
176 static constexpr auto complete{
177 []<class Tag, class... Args>(auto, auto& state, auto& receiver, Tag, Args&&... args) {
178 if constexpr (::std::same_as<Tag, Completion>) {
179 try {
180 let_bind(state, receiver, ::std::forward<Args>(args)...);
181 } catch (...) {
182 ::beman::execution::set_error(::std::move(receiver), ::std::current_exception());
183 }
184 } else {
185 Tag()(::std::move(receiver), ::std::forward<Args>(args)...);
186 }
187 }};
